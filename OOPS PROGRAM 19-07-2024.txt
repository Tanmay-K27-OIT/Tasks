import java.util.Scanner;

public class FunctionalProgram {
    
    
    static boolean isDissarium(int number) {
        String numStr = String.valueOf(number);
        int len = numStr.length();
        int sum = 0;
        
        for (int i = 0; i < len; i++) {
            int digit = Character.getNumericValue(numStr.charAt(i));
            sum += Math.pow(digit, i + 1);
        }
        
        return sum == number;
    }
    
   
    static boolean isVowel(char ch) {
        ch = Character.toLowerCase(ch);
        return ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u';
    }
    
    static boolean isPrime(int number) {
        if (number <= 1) {
            return false;
        }
        
        for (int i = 2; i <= Math.sqrt(number); i++) {
            if (number % i == 0) {
                return false;
            }
        }
        
        return true;
    }
    
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter a number to check if it's Dissarium: ");
        int num1 = scanner.nextInt();
        
        if (isDissarium(num1)) {
            System.out.println(num1 + " is a Dissarium number.");
        } else {
            System.out.println(num1 + " is not a Dissarium number.");
        }
        
        scanner.nextLine();
        
        System.out.print("Enter a character to check if it's a vowel: ");
        char ch = scanner.nextLine().charAt(0);
        
        if (isVowel(ch)) {
            System.out.println(ch + " is a vowel.");
        } else {
            System.out.println(ch + " is not a vowel.");
        }
        
        System.out.print("Enter a number to check if it's prime: ");
        int num2 = scanner.nextInt();
        
        if (isPrime(num2)) {
            System.out.println(num2 + " is a prime number.");
        } else {
            System.out.println(num2 + " is not a prime number.");
        }
        
        scanner.close();
    }
}

import java.util.Scanner;

public class NumberFunctions {
    
 
    static int reverseNumber(int number) {
        int reversedNumber = 0;
        while (number != 0) {
            int digit = number % 10;
            reversedNumber = reversedNumber * 10 + digit;
            number /= 10;
        }
        return reversedNumber;
    }
    
   
    static boolean isArmstrong(int number) {
        int originalNumber = number;
        int sum = 0;
        int digits = String.valueOf(number).length();
        
        while (number != 0) {
            int digit = number % 10;
            sum += Math.pow(digit, digits);
            number /= 10;
        }
        
        return sum == originalNumber;
    }
    

    static boolean isMagic(int number) {
        while (number > 9) {
            number = sumOfDigits(number);
        }
        return number == 1;
    }
    
   
    static int sumOfDigits(int number) {
        int sum = 0;
        while (number != 0) {
            sum += number % 10;
            number /= 10;
        }
        return sum;
    }
    
   
    static int maxOfThree(int num1, int num2, int num3) {
        int max = num1;
        if (num2 > max) {
            max = num2;
        }
        if (num3 > max) {
            max = num3;
        }
        return max;
    }
    
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
      
        System.out.print("Enter a number to reverse: ");
        int numberToReverse = scanner.nextInt();
        int reversedNumber = reverseNumber(numberToReverse);
        System.out.println("Reversed number: " + reversedNumber);
        
      
        System.out.print("\nEnter a number to check if it's Armstrong: ");
        int numberForArmstrong = scanner.nextInt();
        if (isArmstrong(numberForArmstrong)) {
            System.out.println(numberForArmstrong + " is an Armstrong number.");
        } else {
            System.out.println(numberForArmstrong + " is not an Armstrong number.");
        }

          System.out.print("\nEnter a number to check if it's Magic: ");
        int numberForMagic = scanner.nextInt();
        if (isMagic(numberForMagic)) {
            System.out.println(numberForMagic + " is a Magic number.");
        } else {
            System.out.println(numberForMagic + " is not a Magic number.");
        }
        
        
        System.out.print("\nEnter three numbers to find the maximum (separated by spaces): ");
        int num1 = scanner.nextInt();
        int num2 = scanner.nextInt();
        int num3 = scanner.nextInt();
        int max = maxOfThree(num1, num2, num3);
        System.out.println("Maximum of " + num1 + ", " + num2 + ", and " + num3 + " is: " + max);
        
        scanner.close();
    }

import java.util.Scanner;

public class FavoriteFunctions {
    
    
    static int factorial(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        return n * factorial(n - 1);
    }
    
    static boolean isEven(int number) {
        return number % 2 == 0;
    }
    
    static void fibonacci(int n) {
        int a = 0, b = 1;
        System.out.print("Fibonacci series up to " + n + " terms: ");
        for (int i = 0; i < n; i++) {
            System.out.print(a + " ");
            int temp = a + b;
            a = b;
            b = temp;
        }
        System.out.println();
    }
    static int sumOfDigits(int number) {
        int sum = 0;
        while (number != 0) {
            sum += number % 10;
            number /= 10;
        }
        return sum;
    }
    
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Choose one of the favorite functions:");
        System.out.println("1. Factorial calculation");
        System.out.println("2. Check if a number is even");
        System.out.println("3. Generate Fibonacci series");
        System.out.println("4. Sum of digits of a number");
        System.out.print("Enter your choice (1-4): ");
        
        int choice = scanner.nextInt();
        
        switch (choice) {
            case 1:
                System.out.print("Enter a number to calculate factorial: ");
                int numFactorial = scanner.nextInt();
                int factorialResult = factorial(numFactorial);
                System.out.println("Factorial of " + numFactorial + " is: " + factorialResult);
                break;
            case 2:
                System.out.print("Enter a number to check if it's even: ");
                int numEven = scanner.nextInt();
                if (isEven(numEven)) {
                    System.out.println(numEven + " is even.");
                } else {
                    System.out.println(numEven + " is odd.");
                }
                break;
            case 3:
                System.out.print("Enter the number of terms for Fibonacci series: ");
                int numTerms = scanner.nextInt();
                fibonacci(numTerms);
                break;
            case 4:
                System.out.print("Enter a number to find the sum of its digits: ");
                int numSumOfDigits = scanner.nextInt();
                int sumDigitsResult = sumOfDigits(numSumOfDigits);
                System.out.println("Sum of digits of " + numSumOfDigits + " is: " + sumDigitsResult);
                break;
            default:
                System.out.println("Invalid choice. Please enter a number between 1 and 4.");
                break;
        }
        
        scanner.close();
    }
}

import java.util.Scanner;

public class Fact_Demo {
    private int n;
    private int fact = 1;
    

    public void accept(int n) {
        this.n = n;
    }
    
    
    public int cal_fact() {
        for (int i = 1; i <= n; i++) {
            fact *= i;
        }
        return fact;
    }
    
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter the number of values for which you want to calculate factorial: ");
        int numValues = scanner.nextInt();
        
        Fact_Demo[] factObjects = new Fact_Demo[numValues];
        
        
        for (int i = 0; i < numValues; i++) {
            factObjects[i] = new Fact_Demo();
            
            System.out.print("Enter value " + (i + 1) + ": ");
            int value = scanner.nextInt();
            
            factObjects[i].accept(value);
        }
        
        
        System.out.println("\nFactorials:");
        for (int i = 0; i < numValues; i++) {
            int factorial = factObjects[i].cal_fact();
            System.out.println("Factorial of " + factObjects[i].n + " is: " + factorial);
        }
        
        scanner.close();
    }
}
